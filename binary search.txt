def binary_search(numbers, target):
    left = 0
    right = len(numbers) - 1

    while left <= right:
        middle = (left + right) // 2

        if numbers[middle] == target:
            return middle
        elif numbers[middle] < target:
            left = middle + 1
        else:
            right = middle - 1

    return -1


# Example
nums = [1, 3, 5, 7, 9, 11, 13]
result = binary_search(nums, 7)

if result != -1:
    print("Found at index:", result)
else:
    print("Not found")
