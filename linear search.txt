def linear_search(numbers, target):
    for i in range(len(numbers)):
        if numbers[i] == target:
            return i
    return -1


# Example
nums = [4, 2, 7, 1, 9]
result = linear_search(nums, 7)

if result != -1:
    print("Found at index:", result)
else:
    print("Not found")
